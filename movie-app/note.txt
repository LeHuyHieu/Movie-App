Cài đặt bable
npm install --save-dev babel-plugin-module-resolver

/* .babelrc */
{
    "plugins": [
        ["module-resolver", {
            "alias": {
                "@": "./src"
            }
        }]
    ]
}

Editors autocompletion
/* jsconfig.json */

{
    "compilerOptions": {
        "baseUrl": ".",
        "paths": {
            "@/*": ["src/*"]
        }
    }
}
